20 bool XmlParser::parse()

22  QDomDocument dom("mon_xml");
23  QFile file("config.xml");
24  if (!file.open(QIODevice::ReadOnly))
25  return false;

28  file.close();

30  }
31  file.close();
32  QDomElement docElem = dom.documentElement();
33  QDomNode n = docElem.firstChild();
34  while(!n.isNull())
35  {
36  QDomElement e = n.toElement();
37  if(e.tagName()=="MenuConfiguration")
38  {
39  constructMenu(e.firstChild(),"");
40  }
41  else if(e.tagName()=="MediaPlayer")
42  {
43  constructMediaPlayer(e.firstChild());
44  }
45 
46  n = n.nextSibling();
47  }
48  MenuManager* m=dynamic_cast<MenuManager*>(ResourceManager::getInstance().getWidget("Menu Manager") );
49  m->createLayout();
50  return true;
51 }
